    def highlightString : string -> dom = text -> {
        <span class="actionable fadeIn">
            <b> text </b>
        </span>
    }

    def manageNode : dom -> dom = node -> {
        let doClick = node -> node removeChilds 
                                   addChild $ highlightString "Hello World!" in        
            node addChild "Say Hello? " 
                 addChild { highlightString "" onEvent MouseClick doClick }
    }

    document "todo" find map manageNode
